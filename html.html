68(**
69 * Return an array of colors between 'first' and 'second'.
70 * The returned array has the specified 'length'. The colors 'first'
71 * and 'second' are not part of the array.
72 *)
73const func array color: colorScale (in color: first, in color: second,
74 in integer: length) is func
75 result
76 var array color: colorScale is 0 times color.value;
77 local
78 var integer: red_delta is 0;
79 var integer: green_delta is 0;
80 var integer: blue_delta is 0;
81 var integer: number is 0;
82 begin
83 colorScale := length times color.value;
84 red_delta := (second.redLight - first.redLight) div succ(length);
85 green_delta := (second.greenLight - first.greenLight) div succ(length);
86 blue_delta := (second.blueLight - first.blueLight) div succ(length);
87 for number range 1 to length do
88 colorScale[number].redLight := first.redLight + number * red_delta;
89 colorScale[number].greenLight := first.greenLight + number * green_delta;
90 colorScale[number].blueLight := first.blueLight + number * blue_delta;
91 end for;
92 end func;
